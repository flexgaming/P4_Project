import java.util.Scanner;
import java.util.concurrent.*;
import java.util.concurrent.atomic.*;
import java.util.concurrent.CompletableFuture<T>;

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        ExecutorService executor = Executors.newCachedThreadPool();
        int x;
        x = scanner.nextInt();
        if (x > 0) {
            x = x + 1;
            System.out.print("IF statement reached. X = " + x);
        }
        else if (x < 0) {
            System.out.print("x is negative. x = " + x);
        }
        else {
            System.out.print("No IF statement reached. x = " + x);
        }
        executor.shutdown();
        scanner.close();
    }
}
