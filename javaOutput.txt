import java.util.Scanner;
import java.util.concurrent.*;
import java.util.concurrent.locks.*;
import java.util.concurrent.CompletableFuture;
import java.lang.Thread;
import java.util.Arrays;
import java.lang.Math;

public class Main {
    static int counter = 0;
    static int LIMIT = 100000;
    static int[] results = new int[4];
    static int input = 0;
    static Boolean ready = false;
    public static int worker(int start, int end) {
        int i = start;
        int local = 0;
        while (i < end) {
            local = local + i;
            {
                counter = counter + 1;
            }
            i = i + 1;
        }
        return local;
    }
    public static int parallelSum(int n) {
        CompletableFuture<Void> t1 = CompletableFuture.runAsync(() -> {
            try {
                worker(0, n / 4);
            } finally {
            };
        });
        CompletableFuture<Void> t2 = CompletableFuture.runAsync(() -> {
            try {
                worker(n / 4, n / 2);
            } finally {
            };
        });
        CompletableFuture<Void> t3 = CompletableFuture.runAsync(() -> {
            try {
                worker(n / 2, (3 * n) / 4);
            } finally {
            };
        });
        CompletableFuture<Void> t4 = CompletableFuture.runAsync(() -> {
            try {
                worker((3 * n) / 4, n);
            } finally {
            };
        });
        try {
            CompletableFuture.allOf(t1, t2, t3, t4).get();
        } catch (InterruptedException | ExecutionException e) {
            e.printStackTrace();
        }
        return counter;
    }
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        ExecutorService executor = Executors.newCachedThreadPool();
        Lock m0 = new ReentrantLock(); // Shared variable: counter
        System.out.print("Enter a number:");
        input = scanner.nextInt();
        int[] arr = new int[input];
        if (input > LIMIT) {
            System.out.print("Input too large, clamping to LIMIT\n");
            input = LIMIT;
            ready = true;
        }
        else {
            ready = true;
            System.out.print("The input was accepted as is.\n");
        }
        if (ready) {
            for (int i = 0; i < input; i = i + 1) {
                arr[i] = i;
            }
            for (int i = 0; i < input; i = i + 1) {
                System.out.print(arr[i] + "\n");
            }
        }
        System.out.print(counter);
        executor.shutdown();
        scanner.close();
    }
}
