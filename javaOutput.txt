import java.util.Scanner;
import java.util.concurrent.*;
import java.util.concurrent.locks.*;
import java.util.concurrent.CompletableFuture;
import java.lang.Thread;
import java.util.Arrays;
import java.lang.Math;

public class Main {
    static int counter = 1;
    static int LIMIT = 16;
    static int[] results = new int[4];
    static int input = 0;
    static int[] arr = new int[input];
    static Boolean ready = false;
    public static int worker(int start, int end) {
        int i = start;
        int result = counter;
        while (i < end) {
            result = result * arr[i];
            i = i + 1;
        }
        return result;
    }
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        ExecutorService executor = Executors.newCachedThreadPool();
        Lock m0 = new ReentrantLock(); // Shared variable: counter
        System.out.print("Enter a number:");
        input = scanner.nextInt();
        arr = new int[input];
        int index = 0;
        if (input > LIMIT) {
            System.out.print("Input too large, clamping to the LIMIT:" + LIMIT + "\n");
            input = LIMIT;
            ready = true;
        }
        else {
            ready = true;
            System.out.print("The input was accepted as is.\n");
        }
        if (ready) {
            for (int i = 0; i < input; i = i + 1) {
                arr[i] = i + 1;
            }
            while (true) {
                index = index + 1;
                if (index == 67) {
                    System.out.print("SIX SEVEN!!! SIX SEVEN!!! SIX SEVEN!!! SIX SEVEN!!!\n");
                    continue;
                }
                CompletableFuture<Void> t1 = CompletableFuture.runAsync(() -> {
                    {
                        for (double indexer = 100; !m0.tryLock(); indexer = indexer*1.2-((indexer*1.2)%1)) {
                            try {
                                Thread.sleep((long) indexer);
                            } catch (InterruptedException e) {
                                Thread.currentThread().interrupt();
                            }
                        }
                        try {
                            counter = worker(0, input / 4);
                        } finally {
                            m0.unlock();
                        }
                    }
                });
                CompletableFuture<Void> t2 = CompletableFuture.runAsync(() -> {
                    {
                        for (double indexer = 100; !m0.tryLock(); indexer = indexer*1.2-((indexer*1.2)%1)) {
                            try {
                                Thread.sleep((long) indexer);
                            } catch (InterruptedException e) {
                                Thread.currentThread().interrupt();
                            }
                        }
                        try {
                            counter = worker(input / 4, input / 2);
                        } finally {
                            m0.unlock();
                        }
                    }
                });
                CompletableFuture<Void> t3 = CompletableFuture.runAsync(() -> {
                    {
                        for (double indexer = 100; !m0.tryLock(); indexer = indexer*1.2-((indexer*1.2)%1)) {
                            try {
                                Thread.sleep((long) indexer);
                            } catch (InterruptedException e) {
                                Thread.currentThread().interrupt();
                            }
                        }
                        try {
                            counter = worker(input / 2, (3 * input) / 4);
                        } finally {
                            m0.unlock();
                        }
                    }
                });
                CompletableFuture<Void> t4 = CompletableFuture.runAsync(() -> {
                    {
                        for (double indexer = 100; !m0.tryLock(); indexer = indexer*1.2-((indexer*1.2)%1)) {
                            try {
                                Thread.sleep((long) indexer);
                            } catch (InterruptedException e) {
                                Thread.currentThread().interrupt();
                            }
                        }
                        try {
                            counter = worker((3 * input) / 4, input);
                        } finally {
                            m0.unlock();
                        }
                    }
                });
                try {
                    CompletableFuture.allOf(t1, t2, t3, t4).get();
                } catch (InterruptedException | ExecutionException e) {
                    e.printStackTrace();
                }
                System.out.print("The result of " + index + " is: " + counter + "\n");
                counter = 1;
                if (index >= 100) break;

            }
        }
        executor.shutdown();
        scanner.close();
    }
}
